# -*- coding: utf-8 -*-
"""爬取原创力资料网的资料"""
# todo 此程序只实现了抓取pdf、doc、docx等以图片形式预览的文档， PPT格式的文档可用selenium截图
import requests
import os
from bs4 import BeautifulSoup
import re
import threading


class zip118_spider(threading.Thread):
    def __init__(self, p):
        super(zip118_spider, self).__init__()
        """通过文档的cid和p_aid来下载文档"""
        self.session = requests.session()
        self.headers = {
            'Host': 'www.zip118.com',
            'Connection': 'keep-alive',
            'Accept': 'application / json, text / javascript, * / *; q = 0.01',
            'Sec-Fetch-Dest': 'empty',
            'X-Requested-With': 'XMLHttpRequest',
            'User-Agent': 'Mozilla / 5.0(WindowsNT10.0;Win64;x64) AppleWebKit /\
            537.36(KHTML, likeGecko) Chrome / 80.0.3987.122Safari / 537.36',
            'Sec-Fetch-Site': 'same-origin',
            'Sec-Fetch-Mode': 'cors',
            'Referer': 'https://www.zip118.com/205000561_0201/48644.html',
        }
        self.img = ''
        self.sn = 0
        self.NextPage = ''
        self.PageCount = ''
        self.imgLink = ''
        self.img_list = []

        # 获取首页的html, 提取文档的html
        self.init_page = 'https://www.zip118.com/index.php?g=Home&m=View&a=viewDocument&cid=%s&p_aid=%s&status=1' % p
        r = self.session.get(self.init_page, headers=self.headers)
        print('正在下载' + r.json()['data']['title'] + '....')
        print('文件大小:' + r.json()['data']['filesize'])
        self.filename = r.json()['data']['title']
        self.viewUrl = r.json()['viewUrl']
        self.api = {
            'html': 'https:' + self.viewUrl,

        }

        # 从首页的html中提取关键参数f
        str_ = self.viewUrl
        pra = re.findall(r'readpage=(.*?)&furl=(.*?)&n=(.*)', str_)
        self.host = re.findall(r'(.*/)\?', str_)[0]
        self.furl = pra[0][1]
        self.readLimit = pra[0][0]
        r = self.session.get('https:' + self.viewUrl)
        soup = BeautifulSoup(r.text, 'lxml')
        self.f = soup.find('input', id='Url')['value']
        self.getpage = ''

    def run(self):
        self.download_page()  # 获取总页数
        for _ in range(int(self.PageCount) - 1):  # 下载剩余页数
            self.download_page()
        print('%s下载完毕！' % self.filename)

    def download_page(self):
        # print('正在下载第%s张图片' % (self.sn + 1))
        self.getpage = 'https:' + self.host + 'PW/GetPage'
        if not os.path.exists(self.filename):
            os.mkdir(self.filename)
        params = {
            'f': self.f,
            'img': self.img,
            'isMobile': 'false',
            'readLimit': self.readLimit,
            'sn': self.sn,
            'furl': self.furl,
        }

        r = self.session.get(self.getpage, params=params)
        self.NextPage = r.json()['NextPage']  # 图片地址
        self.PageCount = r.json()['PageCount']
        self.sn += 1
        self.img = self.NextPage

        # 保存图片
        self.imgLink = 'https:' + self.host + 'img'
        params = {
            'img': self.img,
            'tp': '',
        }
        r = self.session.get(self.imgLink, params=params)
        path = '%s//%s.jpg' % (self.filename, self.sn)
        with open(path, 'ab') as f:
            f.write(r.content)
        # print('第%s张图片下载完毕' % self.sn)


def get_pra(link):
    headers = {
        'Host': 'www.zip118.com',
        'Connection': 'keep-alive',
        'Accept': 'application / json, text / javascript, * / *; q = 0.01',
        'Sec-Fetch-Dest': 'empty',
        'X-Requested-With': 'XMLHttpRequest',
        'User-Agent': 'Mozilla / 5.0(WindowsNT10.0;Win64;x64) AppleWebKit /\
                537.36(KHTML, likeGecko) Chrome / 80.0.3987.122Safari / 537.36',
    }
    r = requests.get(link, headers=headers)
    soup = BeautifulSoup(r.text, 'lxml')
    pattern = re.compile(r'.*?\.pdf|.*?\.doc|.*?\.docx')  # 仅下载pdf、doc、docx文件
    link_all = soup.find_all('a', class_='son_file_1 son_file_a')
    link_all_ = [i['href'] for i in link_all]
    link = soup.find_all('a', string=pattern)
    link_ = [i['href'] for i in link]
    rm_file = [(i.string, 'https://www.zip118.com' + i['href']) for i in link_all if i not in link]
    [print('文件:%s未被下载(%s)' % i) for i in rm_file]
    print('link')
    
    pattern = re.compile(re.compile(r'/(\d+)_\d+/(\d+)\.html'))
    link_ = [re.findall(pattern, i)[0] for i in link_]
    link_ = [(i[1], i[0]) for i in link_]

if __name__ == '__main__':
    print("--------------原创力资料网下载器:v1.0--------------")
    print("**单个文档仅支持*.doc, *.pdf, *.docx格式的文档下载**\n\n")

    url = input("请输出下载文档地址：")
    # 判断输入的文档地址是单个文档还是多个文档的父链接
    site = re.findall(r'.*?(\d+)_\d+/(\d+)', url)
    if site:
        # 若为单个文档
        thread = zip118_spider((site[0][1], site[0][0]))
        thread.start()
        thread.join()
        exit()

    # 链接为多个文档的父链接， 使用多线程下载文档
    pra = get_pra(url)
    thread = []
    for i in pra:
        thread.append(zip118_spider(i))
    [i.start() for i in thread]
    [i.join() for i in thread]
