多进程共享包含浮点类型数据的列表对象

"""共享包含浮点类型数据的列表对象"""
# -*- coding: utf-8 -*-
from multiprocessing import Process
import multiprocessing
import numpy as np


def fun(num):
    num[:] = np.random.randn(4)


def main():
    val = multiprocessing.Array('f', [1, 2, 3, 4])
    p1 = Process(target=fun, args=(val, ))
    p1.start()
    p1.join()
    print(val[:])


if __name__ == '__main__':
    main()

注意事项

  • 当要对整个列表对象的数据进行改变时，用num[:]
  • 函数的执行程序必须放在，if __name__ == '__main__':这个程序入口
